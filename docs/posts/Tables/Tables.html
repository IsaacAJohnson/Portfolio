Create a summary of BMI by age group

Show the code 
bmi_data_by_age <- NHANES %>%
  mutate(AgeGroup = case_when(
    Age < 18 ~ "Under 18",
    Age >= 18 & Age < 35 ~ "18-34",
    Age >= 35 & Age < 50 ~ "35-49",
    Age >= 50 ~ "50+"
  )) %>%
  group_by(AgeGroup) %>%
  summarize(mean_bmi = mean(BMI, na.rm = TRUE)) %>%
    mutate(mean_bmi = round(mean_bmi, digits = 0))

Add more variables

Show the code 
complex_data_by_age <- NHANES %>%
  mutate(AgeGroup = factor(case_when(
    Age < 18 ~ "Under 18",
    Age >= 18 & Age < 35 ~ "18-34",
    Age >= 35 & Age < 50 ~ "35-49",
    Age >= 50 ~ "50+"
  ), levels = c("Under 18", "18-34", "35-49", "50+"))) %>%
  group_by(AgeGroup) %>%
  summarize(
    mean_bmi = mean(BMI, na.rm = TRUE),
    mean_systolic_BP = mean(BPSysAve, na.rm = TRUE),
    mean_diastolic_BP = mean(BPDiaAve, na.rm = TRUE),
    diabetes_prevalence = mean(Diabetes == "Yes", na.rm = TRUE) * 100
  ) %>%
    mutate(mean_bmi = round(mean_bmi, digits = 0)) %>%
    mutate(mean_systolic_BP = round(mean_systolic_BP, digits = 0)) %>%
    mutate(mean_diastolic_BP = round(mean_diastolic_BP, digits = 0)) %>%
    mutate(diabetes_prevalence = round(diabetes_prevalence, digits = 0))
